def iter1(x,omega):
    n = len(x)
    x[0] = omega*(x[1]-x[n-1])/4+(1-omega)*x[0]
    for i in range(1,n-1):
        x[i] = omega*(x[i-1]+x[i+1])/4+(1-omega)*x[i]
    x[n-1] = omega*(1-x[0]+x[n-2])/4+(1-omega)*x[n-1]
    return x
